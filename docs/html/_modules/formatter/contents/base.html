[docs]class Content(ABC):

    """
    The Implementation defines the interface for all implementation classes.
    """
    def __init__(self, *values) -> None:
        """
        Abstract Content
        """
        self._values = values
        self._link = None

    @property
    def link(self):
        """Returns content link."""
        return self._link

    @link.setter
    def link(self, link):
        """Set content link"""
        self._link = link

    @property
    def values(self):
        """Returns keyword-ed variable."""
        return self._values

    def __str__(self):
        return 'Content:(' + str(self.values) + ')'

        """Returns header level."""
        return self._level
